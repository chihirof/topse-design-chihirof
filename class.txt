interface Command {
    String name
    List<String> params
    exec()
}

    String name
    List<String> params
    exec()
}

    String name
    List<String> params
    exec()
}
note bottom: そのほかのコマンドについても同じようにクラスができる

Command <|-- Echo

interface DirectoryEntry {
    String name
    DirectoryEntry parent
    Date createAt
}

class Directory {
    String name
    DirectoryEntry parent
    Date createAt
    List<DirectoryEntry> childList
    add(DirectoryEntry)
}

class File {
    String name
    DirectoryEntry parent
    Date createAt
}

DirectoryEntry <|-- Directory
DirectoryEntry <|-- File
DirectoryEntry --o Directory
